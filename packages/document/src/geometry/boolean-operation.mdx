## Line

### Lines Intersection

<Playground>
  {
    () => {
      const lineA = [20, 50, 50, 120]
      const lineB = [30, 300, 120, 30]
      const intersect = LinesIntersection(
        [lineA[0], lineA[1]],
        [lineA[2], lineA[3]],
        [lineB[0], lineB[1]],
        [lineB[2], lineB[3]],
      )
      return <>
        <Canvas lines={[lineA, lineB]}/>
        <div>intersect: { intersect ? intersect.join(', ') : 'null' } </div>
      </>
    }
  }
</Playground>

## Polygon

### Is Point In Polygon

<Playground>
  {
    () => {
      const point = [110, 90]
      const polygon = [100, 20, 30, 200, 180, 270]
      const isInPolygon = isPointInPolygon(point, polygon)
      return <>
        <Canvas points={point} polygons={[polygon]}/>
        <div>Point in polygon: { `${isInPolygon}` } </div>
      </>
    }
  }
</Playground>
