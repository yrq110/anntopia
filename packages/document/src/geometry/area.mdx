## VoronoiDiagram

<Playground>
  {
    () => {
      // import { VoronoiDiagram } from '@anntopia/geometry'
      const points = uniform(40)
      let edges = VoronoiDiagram(points)
      edges = edges.map(edge => edge.flat())
      return  <Canvas points={points} polygons={edges}/>
    }
  }
</Playground>

## Shoelace

<Playground>
  {
    () => {
      // import { Chan, Shoelace } from '@anntopia/geometry'
      const points = uniform(40)

      const area = Shoelace(polygon)
      return <>
        <Canvas points={polygon} polygons={[polygon]}/>
        <div>Polygon area: { area } </div>
      </>
    }
  }
</Playground>
