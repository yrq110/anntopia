## DelaunayTriangulation

<Playground>
  {
    () => {
      // import { DelaunayTriangulation } from '@anntopia/geometry'
      const points = uniform(40)
      const triangles = DelaunayTriangulation(points)
      const vertices = []
      triangles.forEach((triangle) => {
        vertices.push(triangle.flat())
      })
      return  <Canvas points={points} polygons={vertices}/>
    }
  }
</Playground>

## MinWeightTriangulation

<Playground>
