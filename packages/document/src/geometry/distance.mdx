## ClosestPair

<Playground>
  {
    () => {
      // import { ClosestPair } from '@anntopia/geometry'
      const points = uniform(10)
      const { distance, begin, end } = ClosestPair(points)
      const pairPoints = [...begin, ...end]
      return <>
        <Canvas points={points} lines={pairPoints} />
        <div>Closest pair points distance: { distance } </div>
      </>
    }
  }
</Playground>
